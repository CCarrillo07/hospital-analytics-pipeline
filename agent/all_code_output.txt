"""
Chart generation helper for the DB AI Agent.

The SQL agent is good for natural language answers.
For charts, we separately ask the LLM to generate a chart-friendly SELECT query,
then we run that query and display the result with Plotly.
"""

import pandas as pd
import plotly.express as px
from sqlalchemy import text

from agent.llm import get_llm
from agent.database import get_engine
from agent.safety import clean_sql_response, is_safe_select_query
from agent.config import ALLOWED_SCHEMA, ALLOWED_TABLES


def generate_chart_sql(question: str) -> str:
    """
    Generate a chart-friendly SQL query from a natural language question.
    """

    llm = get_llm()

    prompt = f"""
You are a PostgreSQL expert.

Create one safe SELECT query for this user question:

{question}

Allowed schema:
{ALLOWED_SCHEMA}

Allowed tables:
{", ".join([f"{ALLOWED_SCHEMA}.{table}" for table in ALLOWED_TABLES])}

Rules:
- Return only SQL.
- Do not use markdown.
- Only use SELECT.
- Always use full table names with the schema prefix, for example {ALLOWED_SCHEMA}.appointments.
- Only query tables in the {ALLOWED_SCHEMA} schema.
- Do not use INSERT, UPDATE, DELETE, DROP, ALTER, TRUNCATE, CREATE, or CALL.
- Use clear column aliases.
- For charts, return two columns when possible:
  1. A category or date column
  2. A numeric metric column
- Limit the result to 50 rows maximum.
"""

    response = llm.invoke(prompt)

    return clean_sql_response(response.content)


def run_chart_sql(sql_query: str) -> pd.DataFrame:
    """
    Run a safe SQL query and return the result as a DataFrame.
    """

    if not is_safe_select_query(sql_query):
        raise ValueError("The generated SQL did not pass safety checks.")

    engine = get_engine()

    with engine.connect() as conn:
        return pd.read_sql(text(sql_query), conn)


def create_chart(df: pd.DataFrame):
    """
    Create a simple Plotly chart from a DataFrame.

    This function assumes the first column is the category/date
    and the second column is the numeric metric.
    """

    if df.empty:
        return None

    if len(df.columns) < 2:
        return None

    x_column = df.columns[0]
    y_column = df.columns[1]

    if not pd.api.types.is_numeric_dtype(df[y_column]):
        return None

    fig = px.bar(
        df,
        x=x_column,
        y=y_column,
        text=y_column,
        title=f"{y_column.replace('_', ' ').title()} by {x_column.replace('_', ' ').title()}"
    )

    fig.update_layout(
        xaxis_title=x_column.replace("_", " ").title(),
        yaxis_title=y_column.replace("_", " ").title()
    )

    return fig

#### .\database.py ####

"""
Database configuration for the DB AI Agent.
"""

from functools import lru_cache

from sqlalchemy import create_engine
from sqlalchemy.engine import Engine

from langchain_community.utilities import SQLDatabase

from agent.config import (
    AI_DB_HOST,
    AI_DB_PORT,
    AI_DB_NAME,
    AI_DB_USER,
    AI_DB_PASSWORD,
    ALLOWED_SCHEMA,
    ALLOWED_TABLES,
)


def get_ai_database_url() -> str:
    """
    Build the PostgreSQL connection URL for the AI agent.
    """

    if not AI_DB_NAME or not AI_DB_USER or not AI_DB_PASSWORD:
        raise ValueError(
            "Missing AI database environment variables. "
            "Please check AI_DB_NAME, AI_DB_USER, and AI_DB_PASSWORD."
        )

    return (
        f"postgresql+psycopg2://{AI_DB_USER}:{AI_DB_PASSWORD}"
        f"@{AI_DB_HOST}:{AI_DB_PORT}/{AI_DB_NAME}"
    )

@lru_cache(maxsize=1)
def get_engine() -> Engine:
    """
    Create a SQLAlchemy engine for direct SQL execution.
    """

    return create_engine(get_ai_database_url())

@lru_cache(maxsize=1)
def get_sql_database() -> SQLDatabase:
    """
    Create the LangChain SQLDatabase object.

    The SQL agent only receives access to the harmonized schema tables.
    """

    return SQLDatabase.from_uri(
        get_ai_database_url(),
        schema=ALLOWED_SCHEMA,
        include_tables=ALLOWED_TABLES,
        sample_rows_in_table_info=3
    )

#### .\safety.py ####

"""
Safety checks for SQL generated by the DB AI Agent.

Important:
The real protection is the read-only PostgreSQL user.
This file adds an extra application-level validation layer.
"""

import re

from agent.config import ALLOWED_SCHEMA


BLOCKED_KEYWORDS = [
    "insert",
    "update",
    "delete",
    "drop",
    "alter",
    "truncate",
    "create",
    "grant",
    "revoke",
    "call",
    "copy",
]


def clean_sql_response(response_text: str) -> str:
    """
    Remove markdown formatting from SQL generated by the model.
    """

    cleaned = response_text.strip()
    cleaned = cleaned.replace("```sql", "")
    cleaned = cleaned.replace("```", "")
    cleaned = cleaned.strip()

    return cleaned


def is_safe_select_query(sql_query: str) -> bool:
    """
    Validate that the SQL query is safe enough to execute.

    Rules:
    - Must start with SELECT
    - Must query the allowed schema
    - Must not contain dangerous SQL keywords
    - Must not contain multiple SQL statements
    """

    normalized_sql = sql_query.lower().strip()

    # Remove one final semicolon if present.
    if normalized_sql.endswith(";"):
        normalized_sql = normalized_sql[:-1].strip()

    # Block multiple statements.
    if ";" in normalized_sql:
        return False

    if not normalized_sql.startswith("select"):
        return False

    if f"{ALLOWED_SCHEMA}." not in normalized_sql:
        return False

    for keyword in BLOCKED_KEYWORDS:
        pattern = rf"\b{keyword}\b"
        if re.search(pattern, normalized_sql):
            return False

    return True

#### .\sql_agent.py ####

"""
LangChain SQL Agent for natural language questions over PostgreSQL.
"""

from functools import lru_cache

from langchain_community.agent_toolkits.sql.toolkit import SQLDatabaseToolkit
from langchain_community.agent_toolkits.sql.base import create_sql_agent

from agent.llm import get_llm
from agent.database import get_sql_database
from agent.config import ALLOWED_SCHEMA, ALLOWED_TABLES


@lru_cache(maxsize=1)
def get_agent_executor():
    """
    Create the LangChain SQL agent executor.

    The agent can inspect table schemas, generate SQL, run SELECT queries,
    and return a natural language answer.
    """

    llm = get_llm()
    db = get_sql_database()

    toolkit = SQLDatabaseToolkit(
        db=db,
        llm=llm
    )

    prefix = f"""
You are a data analytics assistant for a hospital analytics database.

You must answer questions using only the PostgreSQL schema:

{ALLOWED_SCHEMA}

Available tables:

{", ".join(ALLOWED_TABLES)}

Rules:
- Only use SELECT queries.
- Always use full table names with the schema prefix, for example {ALLOWED_SCHEMA}.appointments.
- Only query tables inside the {ALLOWED_SCHEMA} schema.
- Do not query raw tables.
- Do not query automation tables.
- Do not insert, update, delete, drop, truncate, alter, or create anything.
- Never modify the database.
- Be concise and business-friendly.
- When possible, explain the answer in simple business terms.

Very important:
- The sample rows shown in the table schema are only examples.
- Never answer the user's question using only the sample rows.
- If the user asks "how many", "count", "total", "average", "sum", "by", "per", "grouped by", "highest", "lowest", or "most", you must execute a SQL query.
- For aggregation questions, always use GROUP BY when needed.
- Always return the actual result from the database, not only the SQL query.
"""

    return create_sql_agent(
        llm=llm,
        toolkit=toolkit,
        verbose=True,
        handle_parsing_errors=True,
        top_k=20,
        prefix=prefix,
        max_iterations=10
    )


def ask_database(question: str) -> str:
    """
    Ask the SQL agent a natural language question.
    """

    agent_executor = get_agent_executor()

    response = agent_executor.invoke(
        {
            "input": question
        }
    )

    return response.get("output", str(response))
